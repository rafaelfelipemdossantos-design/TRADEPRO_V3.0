const CACHE_NAME = 'tradepro-v3'; const ASSETS = ['./', './index.html', './calculadora.html', './manifest.json']; self.addEventListener('install', (e) => { e.waitUntil(caches.open(CACHE_NAME).then((cache) => cache.addAll(ASSETS))); }); self.addEventListener('fetch', (e) => { e.respondWith(caches.match(e.request).then((res) => res || fetch(e.request))); });
